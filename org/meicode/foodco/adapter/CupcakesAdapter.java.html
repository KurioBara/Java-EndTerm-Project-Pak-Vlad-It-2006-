package org.meicode.foodco.adapter;

import android.content.Context;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.ImageView;
import android.widget.TextView;

import androidx.annotation.NonNull;
import androidx.recyclerview.widget.RecyclerView;

import org.meicode.foodco.R;
import org.meicode.foodco.model.Cupcakes;
import org.meicode.foodco.model.PopularFood;

import java.util.List;

public class CupcakesAdapter extends RecyclerView.Adapter<CupcakesAdapter.CupcakesViewHolder> {

    Context context;
    List<Cupcakes> CupcakesList;



    public CupcakesAdapter(Context context, List<Cupcakes> CupcakesList) {
        this.context = context;
        this.CupcakesList = CupcakesList;
    }

    @NonNull
    @Override
    public CupcakesViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {

        View view = LayoutInflater.from(context).inflate(R.layout.cupcakes_row_item, parent, false);

        return new CupcakesViewHolder(view);
    }

    @Override
    public void onBindViewHolder(@NonNull CupcakesViewHolder holder, int position) {

        holder.foodImage.setImageResource(CupcakesList.get(position).getImageUrl());
        holder.name.setText(CupcakesList.get(position).getName());
        holder.price.setText(CupcakesList.get(position).getPrice());
        holder.rating.setText(CupcakesList.get(position).getRating());

    }

    @Override
    public int getItemCount() {
        return CupcakesList.size();
    }


    public static final class CupcakesViewHolder extends RecyclerView.ViewHolder {

        ImageView foodImage;
        TextView price, name, rating;

        public CupcakesViewHolder(@NonNull View itemView) {
            super(itemView);

            foodImage = itemView.findViewById(R.id.food_image);
            price = itemView.findViewById(R.id.price);
            name = itemView.findViewById(R.id.name);
            rating = itemView.findViewById(R.id.rating);


        }
    }
}
